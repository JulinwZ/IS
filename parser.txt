parser grammar SpaceBattleParser;
options { tokenVocab=SpaceBattleLexer; }

rules: builds
    | orders
    | includes
    | precedes
    ;

builds: ID BUILDS ID (',' ID)*;

includes: ID INCLUDES ID (',' ID)*;
precedes: ID PRECEDES ID;
